import networkx as nx
import matplotlib.pyplot as plt

print(f"Full graph has {G.number_of_nodes()} nodes and {G.number_of_edges()} edges.")

def bfs_sample(graph, start_node, sample_size=500):
    """Return a BFS-based sample subgraph with ~sample_size nodes."""
    visited = set([start_node])
    queue = [start_node]

    while queue and len(visited) < sample_size:
        current = queue.pop(0)
        neighbors = list(graph.neighbors(current))

        for n in neighbors:
            if n not in visited:
                visited.add(n)
                queue.append(n)
            if len(visited) >= sample_size:
                break

    return graph.subgraph(visited).copy()

start_node = list(G.nodes())[0]

visual_subgraph = bfs_sample(G, start_node, sample_size=300)

print(f"Visualization subgraph: {visual_subgraph.number_of_nodes()} nodes, "
      f"{visual_subgraph.number_of_edges()} edges")

plt.figure(figsize=(10, 10))
pos = nx.spring_layout(visual_subgraph, seed=42)

nx.draw_networkx_nodes(
    visual_subgraph, pos,
    node_size=20,
    node_color="skyblue"
)

nx.draw_networkx_edges(
    visual_subgraph, pos,
    alpha=0.3
)

plt.title("BFS Sample Visualization of Spotify User Graph")
plt.axis('off')
plt.show()

nodes = spark.read.csv("/content/nodes.csv", header=True, inferSchema=True)
centralities = spark.read.csv("/content/centrality_with_names.csv", header=True, inferSchema=True)

data = nodes.join(centralities, on="spotify_id", how="inner")

data = data.select(
    "followers",
    "degree_centrality",
    "betweenness_centrality",
    "closeness_centrality",
    "eigenvector_centrality",
    "pagerank"
)

from pyspark.sql.functions import col
data = data.withColumn("followers", col("followers").cast("double"))

data = data.na.drop()

from pyspark.ml.feature import VectorAssembler

assembler = VectorAssembler(
    inputCols=[
        "degree_centrality",
        "betweenness_centrality",
        "closeness_centrality",
        "eigenvector_centrality",
        "pagerank"
    ],
    outputCol="features"
)

final_data = assembler.transform(data)

train_data, test_data = final_data.randomSplit([0.8, 0.2], seed=42)

from pyspark.ml.regression import LinearRegression

lr = LinearRegression(
    labelCol="followers",
    featuresCol="features"
)

model = lr.fit(train_data)

results = model.evaluate(test_data)

print("========== Linear Regression Results ==========")
print("RMSE:", results.rootMeanSquaredError)
print("MAE:", results.meanAbsoluteError)
print("R2 Score:", results.r2)

print("========== Model Coefficients ==========")
print("Intercept:", model.intercept)
print("Coefficients:", model.coefficients)
